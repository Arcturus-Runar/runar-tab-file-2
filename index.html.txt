<script>
const tunings = {
  guitar: { "e": 64, "B": 59, "G": 55, "D": 50, "A": 45, "E": 40 },
  bass:   { "G": 43, "D": 38, "A": 33, "E": 28 }
};

let scheduled = [];

// RÚNAR: bright, high, wide, lots of delay/reverb.
const runar = new Tone.PolySynth(Tone.Synth, {
  oscillator: { type: "triangle" },
  envelope: { attack: 0.16, decay: 0.25, sustain: 0.9, release: 6 }
});
const runarChorus = new Tone.Chorus({ frequency: 1.2, delayTime: 4, depth: 0.8, wet: 0.6 }).start();
const runarDelay = new Tone.FeedbackDelay({ delayTime: "8n.", feedback: 0.45, wet: 0.45 });
const runarReverb = new Tone.Reverb({ decay: 14, wet: 0.72 });
runar.chain(
  new Tone.Filter(700, "highpass"),
  new Tone.Distortion({ distortion: 0.04, wet: 0.12 }),
  runarChorus,
  runarDelay,
  runarReverb,
  new Tone.Volume(-4),
  Tone.Destination
);

// SCOTT: warmer, darker, less wide, holds the harmony.
const scott = new Tone.PolySynth(Tone.Synth, {
  oscillator: { type: "triangle" },
  envelope: { attack: 0.28, decay: 0.35, sustain: 0.82, release: 5 }
});
scott.chain(
  new Tone.Filter(2800, "lowpass"),
  new Tone.Distortion({ distortion: 0.08, wet: 0.16 }),
  new Tone.Reverb({ decay: 7, wet: 0.42 }),
  new Tone.Volume(-12),
  Tone.Destination
);

// CHRIS: low, round, simple.
const chris = new Tone.MonoSynth({
  oscillator: { type: "sine" },
  envelope: { attack: 0.08, decay: 0.4, sustain: 0.8, release: 2.8 }
});
chris.chain(
  new Tone.Filter(900, "lowpass"),
  new Tone.Compressor({ threshold: -20, ratio: 5 }),
  new Tone.Volume(-7),
  Tone.Destination
);

function midiToFreq(midi) {
  return 440 * Math.pow(2, (midi - 69) / 12);
}

function getTempo(text) {
  const match = text.match(/Tempo\s*=\s*(\d+)/i);
  return match ? parseInt(match[1]) : 72;
}

function splitSections(text) {
  const sections = { RUNAR: "", SCOTT: "", CHRIS: "" };
  let current = null;

  text.split("\n").forEach(line => {
    const clean = line.trim().toUpperCase();
    if (clean === "RUNAR" || clean === "SCOTT" || clean === "CHRIS") {
      current = clean;
      return;
    }
    if (current) sections[current] += line + "\n";
  });

  return sections;
}

function parseTab(sectionText, instrument) {
  const lines = sectionText.split("\n").filter(l => l.includes("|"));
  const events = [];

  lines.forEach(line => {
    const label = line.slice(0, line.indexOf("|")).trim();
    const content = line.slice(line.indexOf("|") + 1);
    const base = instrument === "bass" ? tunings.bass[label] : tunings.guitar[label];

    if (base === undefined) return;

    for (let i = 0; i < content.length; i++) {
      if (/\d/.test(content[i])) {
        let fretText = content[i];

        if (/\d/.test(content[i + 1])) {
          fretText += content[i + 1];
          i++;
        }

        const fret = parseInt(fretText);
        const lookahead = content.slice(i + 1, i + 12);

        let midi = base + fret;
        let duration = instrument === "bass" ? 1.1 : 0.65;
        let velocity = instrument === "bass" ? 0.75 : 0.55;

        const sustain = lookahead.match(/~+/);
        if (sustain) duration += sustain[0].length * 0.45;

        const bend = lookahead.match(/b(\d+)/);
        if (bend) {
          midi += parseInt(bend[1]) - fret;
          duration += 0.65;
          velocity += 0.12;
        }

        if (lookahead.includes("h") || lookahead.includes("p")) {
          duration *= 0.75;
          velocity *= 0.85;
        }

        events.push({
          time: i,
          midi,
          duration,
          velocity: Math.min(velocity, 1)
        });
      }
    }
  });

  return events.sort((a, b) => a.time - b.time);
}

async function playTab() {
  await Tone.start();
  stopTab();

  const text = document.getElementById("tabInput").value;
  const tempo = getTempo(text);
  const sections = splitSections(text);

  Tone.Transport.bpm.value = tempo;

  const columnTime = 0.055;

  const allEvents = [
    { synth: runar, events: parseTab(sections.RUNAR, "guitar") },
    { synth: scott, events: parseTab(sections.SCOTT, "guitar") },
    { synth: chris, events: parseTab(sections.CHRIS, "bass") }
  ];

  allEvents.forEach(track => {
    track.events.forEach(ev => {
      const id = Tone.Transport.schedule(time => {
        track.synth.triggerAttackRelease(
          midiToFreq(ev.midi),
          ev.duration,
          time,
          ev.velocity
        );
      }, ev.time * columnTime);
      scheduled.push(id);
    });
  });

  Tone.Transport.start();
}

function stopTab() {
  Tone.Transport.stop();
  Tone.Transport.cancel();
  scheduled = [];
  runar.releaseAll();
  scott.releaseAll();
  chris.triggerRelease();
}
